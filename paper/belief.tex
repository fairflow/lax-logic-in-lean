%% Mono font for the Lean listings: DejaVu covers most of the symbol set;
%% a luaotfload fallback to STIX Two Math fills the rest (⊩, ⊨, ◯ — the
%% listings `literate' table does NOT fire under LuaLaTeX, see unicode.tex).

%% Pandoc artefacts.

%% Listings: Lean source, flexible columns, gentle breaking for the
%% two-column measure.
%% Code blocks are fancyvrb Verbatim under the historical environment name.
%% The listings package is deliberately ABSENT: its scanner reorders
%% non-ASCII glyphs under LuaLaTeX (measured — a `[◯p]` prints `◯[p]`) and
%% its breaklines/literate machinery livelocks; plain verbatim + the mono
%% font with the STIX fallback renders every symbol in order.  Inline code
%% is \texttt with pour-time escaping (no \lstinline anywhere).  The
%% pdflatex/TAPS path, if ever needed, reverts to listings + the literate
%% table kept in unicode.tex.
\DefineVerbatimEnvironment{lstlisting}{Verbatim}{}
\fvset{fontsize=\footnotesize,xleftmargin=0.5em}
%% Long Lean identifiers in prose may break (without hyphen) after an
%% escaped underscore — the column is narrow and the names are not.
\renewcommand{\_}{\textunderscore\allowbreak}
%% Monospace-heavy justified lines: stretch rather than overflow.
\setlength{\emergencystretch}{2em}
